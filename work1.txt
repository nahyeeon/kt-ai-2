def solution(left, right):
    answer = 0
    l_lr=[i for i in range(left,right+1)]
    l_a=[]
    
    for i in range(left,right+1):
        cnt=0
        for k in range(1,i+1):
            if i%k==0:
                cnt=cnt+1
        l_a.append(cnt)
            
    for i in range(len(l_a)):
        if l_a[i]%2==0:
            answer=answer+l_lr[i]
        else:
            answer=answer-l_lr[i]
    return answer
